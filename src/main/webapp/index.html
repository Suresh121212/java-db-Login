from PIL import Image, ImageDraw, ImageFont import secrets background_image = Image.open("C:\\Users\\hp\\OneDrive\\Desktop\\certificate_program\\Certificate_Template.png") certificate = Image.new("RGBA", background_image.size, (680, 1280, 680, 1280)) certificate.paste(background_image, (0, 0)) draw = ImageDraw.Draw(certificate) font_name = ImageFont.truetype('arial.ttf', size=30) name = str(input('Enter the name (ex:- all letter are in capital):')) draw.text((670,310), f' {name}', fill='white', font=font_name, anchor='ms') font_course = ImageFont.truetype('arial.ttf', size=24) course = str(input('Enter the course name (ex:- all letter are in capital):')) draw.text((680,440), f'{course}', fill='green', font=font_course, anchor='ms') font_Date = ImageFont.truetype('arial.ttf', size=20) Date = str(input('Enter the Date (ex:- March 21,2023):')) draw.text((640,555), f'{Date}', fill='white', font=font_Date, anchor='ms') font_code = ImageFont.truetype('arial.ttf', size=20) code = secrets.token_hex(5).upper() draw.text((300,555), f'{code}', fill='white', font=font_code, anchor='ms') certificate.save('C:\\Users\\hp\\OneDrive\\Desktop\\certificate_program\\generated_certificated_data\\Certificate_{}.png'.format(name))
